package org.flowable.engine.test.api.runtime.migration;

import java.io.Serializable;
import java.util.List;

import org.flowable.engine.delegate.DelegateExecution;
import org.flowable.engine.delegate.JavaDelegate;
import org.flowable.engine.impl.context.Context;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ArrayNode;

/**
 * @author martin.grofcik
 */
public class ConvertProcessVariable implements JavaDelegate, Serializable {

    @Override
    public void execute(DelegateExecution execution) {
            List<String> list  = (List<String>) execution.getVariable("listVariable");
            ObjectMapper mapper = Context.getProcessEngineConfiguration().getObjectMapper();
            ArrayNode jsonArray = mapper.createArrayNode();
            list.forEach(listItem ->  jsonArray.add(listItem));
            execution.setVariable("listVariable", jsonArray);
    }
}
