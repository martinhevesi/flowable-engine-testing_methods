    String APP = "app";
    String BPMN = "bpmn";
    String CMMN = "cmmn";
    String DMN = "dmn";
    String EVENT_REGISTRY = "eventRegistry";
    String FORM = "form";
    String PLAN_ITEM = "planItem";
    String TASK = "task";
    String EXTERNAL_WORKER = "externalWorker";

    String BPMN_VARIABLE_AGGREGATION = "bpmnVariableAggregation";
